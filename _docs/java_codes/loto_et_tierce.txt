    private void jButtonValiderNombreLotoActionPerformed(java.awt.event.ActionEvent evt) {

//        String s = "Toto";
        // Le loto
        StringBuilder lsbGrille = new StringBuilder();
        String lsNombre = jTextFieldNombreLoto.getText();
        //String lsMotif = "[0-9]{1,2}";
        String lsMotif = "[1-9]|[1-4][0-9]";
        boolean lbNombre;
        boolean lbExiste;

        jLabelMessage.setText("");

        if (lsNombre.trim().isEmpty()) {
            jLabelMessage.setText("Vous devez saisir !!!");
        } else {
            lbNombre = lsNombre.matches(lsMotif);
            if (!lbNombre) {
                jLabelMessage.setText("Ce n'est pas un nombre ou hors plage [1-49] !!!");
            } else {
                lbExiste = setGrilleLoto.contains(lsNombre);
                if (lbExiste) {
                    jLabelMessage.setText("Ce numéro a déjà été choisi !!!");
                } else {
                    if (iiNumerosSaisis >= MAX_NUMEROS) {
                        jLabelMessage.setText("Le 9 numéros ont été tirés !!!");
                    } else {
                        setGrilleLoto.add(lsNombre);
                        iiNumerosSaisis++;
                        jLabelMessage.setText("Numéros saisis : " + Integer.toString(iiNumerosSaisis) + ". Il en reste encore " + (MAX_NUMEROS - iiNumerosSaisis) + " à saisir !");
                        // Meme desordre
//                        for (Object item : setGrilleLoto) {
//                            lsbGrille.append(item);
//                            lsbGrille.append(" ");
//                        } /// boucle
                        // Meme desordre
                        Iterator it = setGrilleLoto.iterator();
                        while(it.hasNext()) {
                            lsbGrille.append(it.next());
                            lsbGrille.append(" ");
                        }
                        
                        jLabelGrilleLoto.setText(lsbGrille.toString());
                    }
                }
            }
        }
        jTextFieldNombreLoto.requestFocus();
        jTextFieldNombreLoto.setSelectionStart(0);
        jTextFieldNombreLoto.setSelectionEnd(jTextFieldNombreLoto.getText().length());
    }
