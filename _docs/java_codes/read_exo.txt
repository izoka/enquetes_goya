package applifichiers.presentation;

import java.io.*;

// ----------------------------
public class FichierTextLectureExo {

    // -----------------------------------
    public static void main(String[] args) {
        String lsFichier = "editeurs.txt";
        String lsChemin;
        String lsLigne;

        try {
            // --- Ouverture du fichier
            // --- Le fichier est au niveau racine du disque C
            lsChemin = "c:/";
            // --- Le fichier est au niveau racine du projet
            lsChemin = "";
            //lfrFichier = new FileReader(lsChemin);
            // --- Le fichier est dans le dossier des classes !!!
            //lsChemin = "build/classes/packageIntro/";
            // --- Le fichier est dans le dossier Ressources
            //lsChemin = "/___pascal/supports/java_supports/projetsJava/projetIntro/src/packageIntro/ressources/";

            lsChemin += lsFichier;
            System.out.println(lsChemin);

            Reader lfrFichier = new FileReader(lsChemin);
            // --- Bufferisation
            BufferedReader lbrBuffer = new BufferedReader(lfrFichier);

            // --- Lecture des lignes-enregistrements
            while ((lsLigne = lbrBuffer.readLine()) != null) {
			
				//System.out.println(lsLigne);
				
                String[] tChamps = lsLigne.split(";");
                
                if (!lsLigne.trim().equals("")) {
                    for (int i = 0; i < tChamps.length; i++) {
                        String lsChamp = tChamps[i];
                        System.out.println(lsChamp);
                    }
                }
            }
            lbrBuffer.close();
            lfrFichier.close();
        } catch (FileNotFoundException e) {
            System.err.println(e);
        } catch (IOException e) {
            System.err.println(e);
        }
    } /// main

} /// class
